```cpp
#include <bits/stdc++.h>
#define int long long // 定义 int 为 long long 类型，避免溢出
using namespace std;
const int N = 2e5 + 5; // 定义数组最大长度

int a[N]; // t: 测试用例数（未使用），n: 数组长度，q: 查询次数，a: 数组
map<int, int> m; // 用 map 统计每个数值出现的次数

void solve() {
    int n, q;
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        ++m[a[i]]; // 记录 a[i] 出现的次数
    }
    int cnt = 0; // cnt: 严格小于中位数的元素个数
    int lim = n / 2; // 中位数是第 lim + 1 小的数（假设 n >= 1）
    auto it = m.begin(); // 迭代器，从小到大遍历 map
    // 找到第一个使得 n - les - it->second < lim 的 it
    // 即中位数所在的区间
    while (it != m.end() && n - cnt - it->second >= lim) {
        cnt += it->second; // 累加比当前数小的数的个数
        it = next(it); // 移到下一个更大的数
    }
    int p, v; // p: 位置，v: 增加值
    while (q--) {
        cin >> p >> v;
        // 更新前的 a[p] 如果比当前中位数小，则 cnt 需要减 1
        if (a[p] < it->first) --cnt;
        --m[a[p]]; // 减少原数值的计数
        // 更新 a[p]
        a[p] += v;
        // 更新后的 a[p] 如果比当前中位数小，则 cnt 需要加 1
        if (a[p] < it->first) ++cnt;
        ++m[a[p]]; // 增加新数值的计数
        // 动态调整中位数的位置
        while (it != m.end() && n - cnt - it->second >= lim) {
            cnt += it->second; // 扩大严格小于中位数的数的范围
            it = next(it); // 尝试更大的数作为中位数
        } 
        // 输出当前严格小于中位数的元素个数
        cout << cnt << '\n';
    }
    m.clear(); // 清空 map（虽然程序结束时会自动释放）
}

signed main() {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0); // 加速输入输出
    int t;
    cin >> t;
    while (t --) solve();
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
#define int long long
#define pii pair<int,int>
using namespace std;
// 最大房间数
const int N = 2e5 + 5;

// 邻接表存储房间连接关系
vector<int> g[N];
// 记忆化存储：mp[{x,y}]表示从x经过y这条边出发能经过的不同走廊数
// 访问标记：vis[{x,y}]表示是否访问过x到y这条边
map<pii, int> mp, vis;
/*
 * x: 当前房间
 * y: 当前要走的门通向的房间
 * d: 当前已统计的不同走廊数量
 * 返回值：从x经过y这条边出发能经过的不同走廊数
 */
int dfs(int x, int y, int d) {
    // 如果已经计算过这个状态，直接返回结果
    if (mp[{x, y}]) return mp[{x, y}];
    
    // 如果已经访问过这条边，说明形成环路，返回当前计数
    if (vis[{x, y}]) return d;  
    // 标记当前边已访问
    vis[{x, y}] = 1;
    
    // 在y的房间中找到指向x的门，计算下一个要走的门
    int z;
    for (int i = 0; i < g[y].size(); i++) {
        if (g[y][i] == x) {
            z = (i + 1) % g[y].size(); // 循环规则,走当前编号i的下一扇门，如果是末尾就回到第一扇门
            break;
        }
    }
    
    // 递归搜索：如果反向边未被访问过则d+1，否则保持d不变
    if (!vis[{y, x}]) mp[{x, y}] = dfs(y, g[y][z], d + 1);
    else mp[{x, y}] = dfs(y, g[y][z], d);
    
    return mp[{x, y}];
}

signed main() {
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int n;  // 房间数量
    cin >> n; 
    // 输入每个房间的门信息
    for (int i = 1; i <= n; i++) {
        int d, x;  // d:门数量，x:门通向的房间
        cin >> d;
        while (d--) {
            cin >> x;
            g[i].push_back(x);  // 添加到邻接表
        }
         }
    // 计算每个房间的答案
    for (int i = 1; i <= n; i++) {
        vis.clear();  // 清空访问标记
        dfs(i, g[i][0], 0);  // 从第一个门开始搜索
        cout << mp[{i, g[i][0]}] << '\n';  // 输出结果
    }
    
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
#define int long long
#define mod 998244353
using namespace std;

void solve() {
    int n, num;
    cin >> n;  // 读取数组长度n
    
    // 定义动态规划数组：
    // f[i][0]表示前i位以0结尾的方案数
    // f[i][1]表示前i位以1结尾的方案数
    // g[i][0]表示前i位以0结尾时的总块数
    // g[i][1]表示前i位以1结尾时的总块数
    vector<array<int, 2>> f(n + 10), g(n + 10);
    
    // 初始化：
    // 前0位以0结尾的方案数为1（空序列视为以0结尾）
    // 前0位的总块数为0
    f[0][0] = 1, g[0][0] = 0;
    
    for( int i = 1; i <= n; ++ i ) {
        cin >> num;  // 读取当前数字
        
        // 如果当前数字可以是0（即不是1）
        if(num != 1) {
            // 以0结尾的方案数 = 前一位以0或1结尾的方案数之和
            f[i][0] = (f[i - 1][0] + f[i - 1][1]) % mod;
            // 以0结尾的总块数 = 前一位以0或1结尾的总块数之和（因为结尾是0不增加块数）
            g[i][0] = (g[i - 1][0] + g[i - 1][1]) % mod;
        }
        
        // 如果当前数字可以是1（即不是0）
        if(num != 0) {
            // 以1结尾的方案数 = 前一位以0或1结尾的方案数之和
            f[i][1] = (f[i - 1][0] + f[i - 1][1]) % mod;
            // 以1结尾的总块数：
            // - 前一位以0结尾的总块数 + 前一位以0结尾的方案数（因为0->1会新增一个块）
            // - 前一位以1结尾的总块数（因为1->1不会新增块）
            g[i][1] = (g[i - 1][0] + g[i - 1][1] + f[i - 1][0]) % mod;
        }
    }
    
    // 最终结果是所有可能情况的总块数之和（以0或1结尾）
    cout << (g[n][1] + g[n][0]) % mod << "\n";
}

signed main() {
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);  // 加速输入输出
    int t;
    cin >> t;  // 读取测试用例数量
    while(t--) solve();  // 处理每个测试用例
    return 0;
}
```

```cpp
        a=a*a%mod;
        b>>=1;
```

```cpp
#include <bits/stdc++.h>
#define int long long
#define inf 1e18
using namespace std;

void solve() {
    // 输入物品数量n和背包容量v，初始化答案ans为0
    int n, v;
    cin >> n >> v;
    
    // 定义物品数组，每个物品有三个属性：h(价值), s(体积), d(衰减系数)
    // 初始化为全0，数组大小为n+5以避免边界问题
    vector<array<int, 3>> a(n + 5);
    
    // 输入每个物品的属性
    for( int i = 0; i < n; ++ i ) {
        int h, s, d;
        cin >> h >> s >> d;
        a[i] = {h, s, d};  // 存储物品属性
    }
    int ans = 0;
    // 动态规划求解：
    // 当n>500时，直接从v开始枚举（优化策略）
    // 否则从0开始枚举所有可能的剩余容量S
    for( int S = ((n > 500) ? v : 0); S <= v; ++ S ) {
        // 初始化dp数组，dp[j]表示容量为j时的最大价值
        // 初始值为-inf（表示不可达），dp[0]=0（容量为0时价值为0）
        vector<int> dp(S + 1, -inf);
        dp[0] = 0;
        
        // 遍历所有物品
        for( int i = 0; i < n; ++ i ) {
            // 获取当前物品的属性
            int h = a[i][0];  // 物品价值
            int s = a[i][1];  // 物品体积
            int d = a[i][2];  // 衰减系数
            
            // 01背包逆序更新
            for( int j = S; j >= s; -- j ) {
                // 状态转移方程：考虑物品i放入背包的情况
                // 价值计算包含衰减项：d*(v-S)
                dp[j] = max(dp[j], dp[j - s] + h - d * (v - S));
            }
        }
        // 更新全局最大价值
        ans = max(dp.back(), ans);
    }
    // 输出结果
    cout << ans << "\n";
}

signed main() {
    // 加速输入输出
    ios::sync_with_stdio(0), cin.tie(0);
    
    // 处理多个测试用例
    int t;
    cin >> t;
    while(t --) solve();
    
    return 0;
}
```
